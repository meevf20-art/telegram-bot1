from telegram import Update
from telegram.ext import ApplicationBuilder, MessageHandler, filters, ContextTypes
import requests

BOT_TOKEN = "PUT_TELEGRAM_BOT_TOKEN"
API_KEY = "PUT_XIAOMI_API_KEY"

API_URL = "https://token-plan-sgp.xiaomimimo.com/v1/chat/completions"

memory = {}

def add_memory(user_id, role, content):
    memory.setdefault(user_id, [])
    memory[user_id].append({"role": role, "content": content})
    memory[user_id] = memory[user_id][-10:]

def ask_ai(messages):
    res = requests.post(API_URL, json={
        "model": "mimo-v2.5-pro",
        "messages": messages,
        "temperature": 0.7
    }, headers={"api-key": API_KEY})

    return res.json()["choices"][0]["message"]["content"]

async def handle(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.message.chat_id
    text = update.message.text

    add_memory(user_id, "user", text)

    messages = [
        {"role": "system", "content": "أنت مساعد ذكي مفيد."}
    ]

    messages += memory[user_id]

    reply = ask_ai(messages)

    add_memory(user_id, "assistant", reply)

    await update.message.reply_text(reply)

app = ApplicationBuilder().token(BOT_TOKEN).build()
app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle))

app.run_polling()
